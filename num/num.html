  2 # The total number of certicates in two solutions
  3 total_num_dttm <- function(N, beta = 0.3, p = 0.3, k = 8)
  4 {
  5 
  6     num <- N * 100 / (beta * k) + (N * (k - 1)) + (beta / 100) * p * N^2 + k^2 * (k - 1) + (beta / 100) * p * N^2
  7 
  8     return (num)
  9 }
 10 
 11 total_num_ttm <- function(N, beta = 0.3, p = 0.3, k = 8)
 12 {
 13     num <- (N * 100 / beta) + (2 * beta / 100) * p * N^2
 14 
 15     return(num)
 16 }
 17 
 18 # The number of certificates hold by single normal node
 19 num_dttm_single_normal_node <- function(N, beta = 0.3, p = 0.3, k = 8)
 20 {
 21     num <- 100 / (beta * k) + (k - 1) + (p * N * beta / 100)
 22 
 23     return (num)
 24 }
 25 
 26 num_ttm_single_normal_node <- function(N, beta = 0.3, p = 0.3, k = 8)
 27 {
 28     num <- (100 / beta) + (p * N * beta / 100)
 29 
 30     return(num)
 31 }
 32 
 33 # The number of certificates hold by single hub node
 34 num_dttm_single_hub_node <- function(N, beta = 0.3, p = 0.3, k = 8)
 35 {
 36     num <- (k - 1) + (p * N / k)
 37 
 38     return (num)
 39 }
 40 
 41 
 42 num_ttm_single_hub_node <- function(N, beta = 0.3, p = 0.3, k = 8)
 43 {
 44     num <- (p * N)
 45 
 46     return (num)
 47 }

 56 plot_total_num <- function()
 57 {
 58     x1 <- vector()
 59     y1 <- vector()
 60 
 61     x2 <- vector()
 62     y2 <- vector()
 63 
 64     index = 25000
 65     i = 1
 66     while(index <= 65000)
 67     {
 68         print(total_num_dttm(index))
 69 
 70         x1[i] <- index
 71         y1[i] <- total_num_dttm(index)
 72 
 73         x2[i] <- index
 74         y2[i] <- total_num_ttm(index)
 75 
 76         i = i + 1
 77         index = index + 5000
 78     }
 79 
 80     # Calculate range from min to max value of y1 and y2
 81     g_range <- range(y1, y2)
 82     print("g_range")
 83     print(g_range)
 84 
 85     print("x1")
 86     print(x1)
 87     print("y1")
 88     print(y1)
 89     print("y2")
 90     print(y2)
 91 
 92     # Graph autos using y axis that ranges from 0 to max value in y1 or y2 vector.  
 93     # Turn off axes and annotations (axis labels) so we can specify them ourself
 94     plot(y1, type = "b", col = "blue", ylim = g_range, axes = FALSE, ann = FALSE)
 95 
 96     x_coordinate <- vector()
 97 
 98     for(index in 1:length(x1))
 99     {
100         x_coordinate[index] <- x1[index] / 1000
101     }
102 
103     axis(1, at = 1:length(x_coordinate), labels = x_coordinate)
104 
105     # Make y axis with horizontal labels that display ticks at every 4 marks. 4*0:g_range[2] is equivalent to c(0,4,8,12).
106     axis(2, las = 1, at = 10000000*0:g_range[2])
107 
108     # Create box around plot
109     box()
110 
111     # Graph y2 with red dashed line and square points
112     lines(y2, type = "o", pch = 22, lty = 2, col = "red")
113 
114     # Create a title with a red, bold/italic font
115     title(main = "The total number of certicates in two solutions", col.main = "red", font.main = 4)
116 
117     # Label the x and y axes with dark green text
118     title(xlab = "N (K)", col.lab = rgb(0, 0.5, 0))
119     title(ylab = "Num (K)", col.lab = rgb(0, 0.5, 0))
120 
121     # Create a legend at (25000, g_range[2]) that is slightly smaller (cex) and uses the same line colors and points used by the actual plots
122     legend(25000, g_range[2], c("total_num_dttm", "total_num_ttm"), cex = 0.7, col = c("blue", "red"), pch = 21:22, lty = 1:2);
123 }
